36 /* Registers decorated load/store addresses */
37 #define MMAU__X0 0xF0004000l /* Accumulator register X0 */
38 #define MMAU__X1 0xF0004004l /* Accumulator register X1 */
39 #define MMAU__X2 0xF0004008l /* Accumulator register X2 */
40 #define MMAU__X3 0xF000400Cl /* Accumulator register X3 */
41 #define MMAU__A0 0xF0004010l /* Accumulator register A0 */
42 #define MMAU__A1 0xF0004014l /* Accumulator register A1 */
43 #define MMAU__A10 0xF0004010l /* Accumulator register pair A10 */
44 /* Unsigned integer instructions decorated load/store addresses */
45 #define MMAU__REGRW 0xf0004000l /* Registers RW */
46 #define MMAU__UMUL 0xf0004020l /* A10=X2*X3 */
47 #define MMAU__UMULD 0xf0004040l /* A10=X21*X3 */
48 #define MMAU__UMULDA 0xf0004060l /* A10=A10*X3 */
49 #define MMAU__UMAC 0xf00040A0l /* A10=X2*X3+A10 */
50 #define MMAU__UMACD 0xf00040C0l /* A10=X21*X3+A10 */
51 #define MMAU__UMACDA 0xf00040E0l /* A10=A10*X3+X21 */
52 #define MMAU__UDIV 0xf0004120l /* X21/X3=A10 */
53 #define MMAU__UDIVD 0xf0004140l /* A10=X2/X3 */
54 #define MMAU__UDIVDA 0xf0004160l /* A10=X21/X3 */
55 #define MMAU__UDIVDD 0xf0004180l /* A10=A10/X3 */
56 #define MMAU__UDIVDDA 0xf00041A0l /* A10=A10/X32 */
57 #define MMAU__USQR 0xf0004220l /* A10=SQR(X3) */
58 #define MMAU__USQRD 0xf0004240l /* A10=SQR(X32) */
59 #define MMAU__USQRDA 0xf0004260l /* A10=SQR(A10) */
60 /* Signed fractional instructions decorated load/store addresses */
61 #define MMAU__QSQR 0xf00042A0l /* A10=SQR(X3) */
62 #define MMAU__QSQRD 0xf00042C0l /* A10=SQR(X32) */
63 #define MMAU__QSQRDA 0xf00042E0l /* A10=SQR(A10) */
64 #define MMAU__QDIV 0xf0004320l /* A10=X2/X3 */
65 #define MMAU__QDIVD 0xf0004340l /* A10=X21/X3 */
66 #define MMAU__QDIVDA 0xf0004360l /* A10=A10/X3 */
67 #define MMAU__QMUL 0xf0004420l /* A10=X2*X3 */
68 #define MMAU__QMULD 0xf0004440l /* A10=X21*X3 */
69 #define MMAU__QMULDA 0xf0004460l /* A10=A10*X3 */
70 #define MMAU__QMAC 0xf00044A0l /* A10=X2*X3+A10 */
71 #define MMAU__QMACD 0xf00044C0l /* A10=X21*X3+A10 */
72 #define MMAU__QMACDA 0xf00044E0l /* A10=A10*X3+X21 */
73 /* Signed integer instructions decorated load/store addresses */
74 #define MMAU__SMUL 0xf0004620l /* A10=X2*X3 */
75 #define MMAU__SMULD 0xf0004640l /* A10=X21*X3 */
76 #define MMAU__SMULDA 0xf0004660l /* A10=A10*X3 */
77 #define MMAU__SMAC 0xf00046A0l /* A10=X2*X3+A10 */
78 #define MMAU__SMACD 0xf00046C0l /* A10=X21*X3+A10 */
79 #define MMAU__SMACDA 0xf00046E0l /* A10=A10*X3+X21 */
80 #define MMAU__SDIV 0xf0004720l /* A10=X2/X3 */
81 #define MMAU__SDIVD 0xf0004740l /* A10=X21/X3 */
82 #define MMAU__SDIVDA 0xf0004760l /* A10=A10/X3 */
83 #define MMAU__SDIVDD 0xf0004780l /* A10=X10/X32 */
84 #define MMAU__SDIVDDA 0xf00047A0l /* A10=A10/X32 */
85 /* Auxiliary decorated load/store addresses */
86 #define MMAU__SAT 0xf0004800l /* Saturation */
