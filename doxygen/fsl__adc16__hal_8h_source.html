285 typedef struct Adc16HwCmpConfig
286 {
287  bool hwCmpEnable;
288  bool hwCmpGreaterThanEnable;
289  /*
290  false - Configures less than the threshold. The outside and inside range are not inclusive.
291  The functionality is based on the values
292  placed in CV1 and CV2.
293  true - Configures greater than or equal to the threshold. The outside and inside
294  ranges are inclusive. The functionality is based on the values placed in
295  CV1 and CV2.
296  */
297  bool hwCmpRangeEnable;
298  /*
299  Configures the comparator function to check if the conversion result of the
300  input being monitored is either between or outside the range formed by
301  CV1 and CV2 and determined by the value of hwCmpGreaterThanEnable.
302 
303  false - Range function disabled. Only CV1 is compared.
304  true - Range function enabled. Both CV1 and CV2 are compared.
305  */
306  uint16_t cmpValue1;
307  uint16_t cmpValue2;
308 } adc16_hw_cmp_config_t;
