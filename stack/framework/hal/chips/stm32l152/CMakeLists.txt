#Add the linker script to use to the linker flags

#Add a 'global' definition for the specific MCU
EXPORT_GLOBAL_COMPILE_DEFINITIONS("-DSTM32L152xE")


SET(LINKER_SCRIPT "${CMAKE_CURRENT_SOURCE_DIR}/linker/stm32l152re_flash.ld" CACHE FILEPATH "")

#SET(LINKER_FLAGS "-Xlinker  -Map=.map" CACHE STRING "")

ENABLE_LANGUAGE(ASM)
SET_PROPERTY(SOURCE Drivers/CMSIS/Device/ST/STM32L1xx/Source/Templates/gcc/startup_stm32l152xe.s PROPERTY LANGUAGE ASM)
INCLUDE_DIRECTORIES(
        Drivers/CMSIS/Include
        Drivers/CMSIS/Device/ST/STM32L1xx/Include
        Drivers/STM32L1xx_HAL_Driver/Inc
)

##Export the 'inc' directory globally		    
EXPORT_GLOBAL_INCLUDE_DIRECTORIES(
        inc
        Drivers/CMSIS/Include
        Drivers/CMSIS/Device/ST/STM32L1xx/Include
        Drivers/STM32L1xx_HAL_Driver/Inc
)

#Please note we include 'inc' here since for complicated cmake reasons directories exported 
#with 'GLOBAL_INCLUDE_DIRECTORIES'from chip directories are not exported to the platform directory
EXPORT_PLATFORM_INCLUDE_DIRECTORIES(
        Drivers/CMSIS/Include
        Drivers/CMSIS/Device/ST/STM32L1xx/Include
        Drivers/STM32L1xx_HAL_Driver/Inc
)

#An object library with name '${CHIP_LIBRARY_NAME}' MUST be generated by th CMakeLists.txt file for every chip
ADD_LIBRARY (${CHIP_LIBRARY_NAME} OBJECT
        Drivers/CMSIS/Device/ST/STM32L1xx/Source/Templates/gcc/startup_stm32l152xe.s
        Drivers/CMSIS/Device/ST/STM32L1xx/Source/Templates/system_stm32l1xx.c
        Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_gpio.c
	    stm32l152_atomic.c
	    stm32l152_system.c
	    stm32l152_timer.c
	    stm32l152_mcu.c
	    stm32l152_pins.c
        stm32l152_watchdog.c
        stm32l152_gpio.c
)
